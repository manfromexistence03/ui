Add the following animations to your `tailwind.config.js` file:

```js title="tailwind.config.js" {5-18}
/** @type {import('tailwindcss').Config} */
module.exports = {
  theme: {
    extend: {
        keyframes: {
            'spinner-leaf-fade': {
                '0%, 100%': { opacity: '0' },
                '50%': { opacity: '1' },
            },
        },
        animation: {
            'spinner-leaf-fade': 'spinner-leaf-fade 800ms linear infinite',
        }
    },
  },
}
```
